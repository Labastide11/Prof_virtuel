// À intégrer dans Code.gs si l'action enregistrer_parcours n'existe pas encore.

function enregistrerParcours_(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheetName = 'parcours_eleves';
  let sh = ss.getSheetByName(sheetName);
  if (!sh) sh = ss.insertSheet(sheetName);

  const headers = [
    'date','heure','prenom','type','matiere','domaine','titre','detail',
    'score','total','niveau','temps_secondes','source','appareil','id_evenement'
  ];

  const firstRow = sh.getRange(1, 1, 1, headers.length).getValues()[0];
  const needsHeaders = firstRow.every(v => String(v || '').trim() === '');
  if (needsHeaders) sh.getRange(1, 1, 1, headers.length).setValues([headers]);

  const p = e.parameter || {};
  const id = String(p.id_evenement || '').trim();

  // Anti-doublon simple sur id_evenement
  if (id && sh.getLastRow() > 1) {
    const ids = sh.getRange(2, 15, sh.getLastRow() - 1, 1).getValues().flat().map(String);
    if (ids.includes(id)) return { ok:true, sheet:sheetName, duplicate:true, id_evenement:id };
  }

  const row = headers.map(h => p[h] || '');
  sh.appendRow(row);
  return { ok:true, sheet:sheetName, added:true, id_evenement:id };
}
